threeSum

var arr = [-1, 0, 1, 2, -1, -4]; // Output: [ [ -1, -1, 2 ], [ -1, 0, 1 ] ] var threeSum = function(nums) { nums.sort((a, b) => a - b); let answer = []; if (nums.length < 3) { return answer; } for (let i = 0; i < nums.length; i++) { if (nums[i] > 0) { break; } if (i > 0 && nums[i] === nums[i - 1]) { continue; } let low = i + 1; let high = nums.length - 1; while (low < high) { const sum = nums[i] + nums[low] + nums[high]; if (sum > 0) { high--; } else if (sum < 0) { low++; } else { answer.push([nums[i], nums[low], nums[high]]); let lastLowOccurrence = nums[low]; let lastHighOccurrence = nums[high]; while (low < high && nums[low] === lastLowOccurrence) { low++; } while (low < high && nums[high] === lastHighOccurrence) { high--; } } } } return answer; };

Remove Duplicates from Sorted Array

var arr = [0, 0, 1, 1, 7, 8, 7, 1, 2, 2, 3, 3, 4]; function removeDuplicates(arr) { // First, sort the array arr.sort(); let y = 1; let res = []; // Initialize the first element of the result array res[0] = arr[0]; // Iterate through the array to remove duplicates for (let x = 1; x < arr.length; x++) { // If the current element is different from the previous one, add it to the result array if (arr[x] != arr[x - 1]) { res[y] = arr[x]; y++; } } return res; } console.log(removeDuplicates(arr)); // The Intuition is to use two pointers, x and y, to iterate through the array. // The variable y is used to keep track of the current index where a unique element should be placed. // The initial value of y is 1 since the first element in the array is always unique // and doesn't need to be changed.

BubbleSort

const list = [15, 5, 20, 35, 2, 42, 67, 17]; // n = 8; no. of elements in an array // In bubble sorting, a[i] is compared with a[i+1], // and if these two elements are in the wrong order then we have to swap those elements. // If the elements are in the correct order then no swapping will be done. function bubbleSortArray(arr) { var temp = null; var flag = null; for (var x = 0; x < arr.length - 1; x++) { flag = 0; for (var y = 0; y < arr.length - 1 - x; y++) { if (arr[y] > arr[y + 1]) { temp = arr[y]; arr[y] = arr[y + 1]; arr[y + 1] = temp; flag = 1; } } if (flag === 0) { break; } } return arr; } console.log(bubbleSortArray(list)); // [ 2, 5, 15, 17, 20, 35, 42, 67] // How to find out time complexity, we have to check how many for loops are there. // Eg 2. Then, to find out the time complexity you have to check how many calculations are there, // how many computations are there within the inner for loop. // Best case is array is already sorted, (O(n) - order of n) // Worst complexity, the time complexity will be (O(n^2) - Order of Nsquare)

Binery Search - Search ele from an Array

// Binary search always works in a sorted array // Here we divide the array into two halves and search. // Case 1: data === arr[midIndex] // Case 2: data < arr[midIndex] // Case 3: data > arr[midIndex] // data: data you want to search // l: left index // r: right index // mid: mid index function binarySearch(arr, n, data) { let l = 0; let r = n - 1; // here we check if l <= r, if r > l, then the element does not exist, then return -1. while (l <= r) { let mid = Math.floor((l + r) / 2); if (data === arr[mid]) { return mid; } else if (data < arr[mid]) { r = mid - 1; } else { l = mid + 1; } } return -1; } const arr = [5, 9, 17, 23, 25, 45, 59, 63, 71, 89]; const data = 59; console.log(binarySearch(arr, arr.length, 59)); /* What is the time complexity of this algorithm? We are reducing the search space by half, in such case the time complexity will be O(logn) - Order of log n, and the best case will be O(1) - Order of 1, if the search data is 25. */

Insertion Sort

/* In this technique, the given array is divided into two parts: Sorted sublist and Unsorted sublist. We take one value of the Unsorted sublist and identify an appropriate place for that value in the sorted sublist. After finding the appropriate place in the sorted sublist, we will insert that value into the sorted sublist. Initially, arr[i] is the sorted sublist and from arr[i+1] until the end of the array is the Unsorted sublist.*/ const list = [5, 4, 10, 1, 6, 2]; function insertionSort(arr) { var temp = null; for (let i = 1; i < arr.length; i++) { temp = arr[i]; let j = i - 1; while (j >= 0 && arr[j] > temp) { arr[j + 1] = arr[j]; j--; } arr[j + 1] = temp; } return arr; } console.log(insertionSort(list)); // Output: [1, 2, 4, 5, 6, 10] /* Time complexity in the worst case scenario, when an array is given in descending order and you have to sort that array in ascending order, in that case, the time complexity will be O(n^2) - Order of Nsquare. Best case scenario, the time complexity will be O(n) - Order of n. When the array is already in ascending order. */
